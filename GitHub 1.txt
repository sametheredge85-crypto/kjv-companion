        // ============== SEARCH & DISPLAY ==============
        function getHarmony() {
            const input = document.getElementById('verse-input').value.toLowerCase().trim();
            const result = kjvData[input];
            const resultDiv = document.getElementById('harmony-result');
            
            if (result) {
                resultDiv.innerHTML = `
                    <div class="harmony-result">
                        <div class="verse-text">${result.verse}</div>
                        <div class="section-title">Context:</div>
                        <p>${result.context}</p>
                        <div class="section-title">Harmony of Scripture:</div>
                        <ul style="padding-left:20px;">
                            ${result.harmony.map(h => `<li>${h}</li>`).join('')}
                        </ul>
                        <div class="section-title">Insights:</div>
                        <p>${result.insights}</p>
                        <div class="section-title">Application:</div>
                        <p style="font-style:italic;">${result.application}</p>
                    </div>
                `;
                resultDiv.style.display = 'block';
            } else {
                resultDiv.innerHTML = `
                    <div class="card">
                        <p style="text-align:center; color:#8b5a2b;">
                            No harmony found for "${input}". Try: <strong>john 3:16</strong>, <strong>easter</strong>, <strong>wisdom</strong>, <strong>proverbs 8:10-12</strong>, or <strong>james 1:5</strong>
                        </p>
                    </div>
                `;
                resultDiv.style.display = 'block';
            }
        }

                    </div>
                `;

                    <div class="card">

                            Search the 7 Sub-Annexes for your topic. Try searching for: <strong>wisdom and the heart</strong>
